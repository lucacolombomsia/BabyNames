    def Count(self,state="",year="") :
        """Return count of baby names for given state and year.
        The function can accept a single state or a list of states.
        Same for year: it accepts one year or a list of years."""
        df = self.data
        
        if ((type(state) == str) & (state!= "")) : state = [state]
        if type(year) == int : year = [year] 
        
        if state != "" :
            df = df[df.state.isin(state)]
        if year != "" :
            df = df[df.year.isin(year)]
        
        return df.births.sum()
    
    
    def Top10BabyNames(self, state="IL", year=2015):
        """Return top 10 baby names, both male and female, for given state and year.
        The function can accept a single state or a list of states.
        Same for year: it accepts one year or a list of years."""
        def get_top10(group):
            return group.sort_values(by='births',ascending=False).head(10)
        
        df = self.data
        
        if ((type(state) == str) & (state!= "")) : state = [state]
        if type(year) == int : year = [year] 
            
        if state != "" :
            df = df[df.state.isin(state)]
        if year != "" :
            df = df[df.year.isin(year)]
        
        df = df[["name", "births", "sex"]].groupby(["name", "sex"], as_index = False
                                                      ).sum().groupby("sex", as_index = False
                                                                     ).apply(get_top10)[["name","sex"]]
        df["Rank"] = list(range(1,11))*2
        df = df.pivot(index = "Rank", columns = "sex", values = "name").reset_index()
        result = pd.DataFrame({"Rank" : df.Rank, "Male" : df.M, "Female" : df.F})[["Rank", "Male", "Female"]]
        return result
    
    
    def ChangeOfPopularity(self, fromYear=2014, toYear=2015, top=10):
        """The function, for a given date range, lists the most popular names that increased in popularity,
        decreased in popularity and maintained the same popularity (rounded at 4 decimals)."""
        def get_top(group):
            return group.sort_values(by=toYear,ascending=False).head(top)
        def add_prop(group):
            group['prop'] = round(100.0*group.births / group.births.sum(), 4)
            return group
        
        df = self.data[(self.data["year"] == fromYear) | (self.data["year"] == toYear)
                      ].groupby(["name", "year"], as_index = False).sum().groupby("year", 
                                                                                  as_index = False).apply(add_prop)
        df = df.pivot(index = "name", columns = "year", values = "prop").fillna(0).reset_index()

        df["delta"] = df[toYear] - df[fromYear]
        df["sign"] = "neg" 
        df.loc[(df.delta>0),"sign"] = "pos"
        df.loc[(df.delta==0),"sign"] = "zero"
        df = df.groupby("sign").apply(get_top)[["sign", "name", toYear]].reset_index(drop = True)
        df["Rank"] = list(range(1,top+1))*3

        df = df.pivot(index = "Rank", columns="sign", values = "name")
        result = pd.DataFrame({"c1" : df.pos, "c2" : df.neg, "c3" : df.zero}
                             ).rename(columns={"c1": "More popular", "c2" : "Less popular", "c3" : "Same popularity"})
        return result
    
    
    def Top5NamesPerYear(self,year=2015,sex=''):
        """Returns the five most frequent names, by State, for male, female, or both in a given year.
        It accepts both a single year or a list of years."""
        sex = ["M", "F"] if sex == '' else [sex]
        year = self.years_list if year == '' else [year]
        
        cols = ["State"]
        for x in range(1,6):
            cols.append("Rank {}".format(x))
            cols.append("Num")
        top5df = pd.DataFrame(columns=cols, index=range(51))

        for i in range(len(self.states_list)):
            df = self.data[(self.data["year"].isin(year)) & (self.data["sex"].isin(sex)) & 
                          (self.data["state"]==self.states_list[i])
                          ].sort_values(by="births", ascending = False).head(5).reset_index(drop = True)

            top5df.iloc[i,0] = self.us_states[self.states_list[i]]
            for x in range(5):
                top5df.iloc[i,x*2+1] = df.iloc[x,3]
                top5df.iloc[i,x*2+2] = df.iloc[x,4]
        return top5df
    
    
    def NamePopularityPlot(self, name='Jim', yearRange=(2000,2015), state='IL', sex='M'):
        """Plots changes in popularity for a given name of a given sex in a given state over a given date range.
        Setting state="" and sex="" will respectively perform the analysis across all states and for both sexes.
        If a name is missing in a given year for the chosen state-sex combination, then popularity is set to 0.""" 
        def add_prop(group):
            group['prop'] = round(100.0*group.births / group.births.sum(), 4)
            return group
        
        if yearRange == "" : yearRange = (min(self.years_list), max(self.years_list)) 

        idx = pd.DataFrame({"year" : range(yearRange[0], yearRange[1]+1)})
        
        df = self.data
        if ((type(state) == str) & (state!= "")) : state = [state]
        if ((type(sex) == str) & (sex!= "")) : sex = [sex]
        
        if state != "" :
            df = df[df.state.isin(state)]
        if sex != "" :
            df = df[df.sex.isin(sex)]
            
        df = df.groupby(["name","year"],as_index=False).sum().groupby('year').apply(add_prop)
        df = df[(df["year"]>=yearRange[0]) & (df["year"]<=yearRange[1]) &
               (df.name == name)][["prop", "year"]]
        idx = idx.merge(df, how="left", on="year").fillna(0)

        t = "Popularity of name {} from {} to {}".format(name, yearRange[0], yearRange[1])
        idx.set_index("year")["prop"].plot(title=t,figsize=(12,6))
        

    def NameFlip(self, n=10):
        """For every name in every year, compute the delta between female and male births 
        (positive delta indicates female prevalence). Then for every name, take the maximum and
        minimum delta over all years. If the max delta is positive and the min delta is negative,
        then the name flipped from male prevalence to female prevalence or viceversa.
        Finally, out of all the names that flipped, plot the ones that have been the most popular in
        terms of total births, looking only at the sex for which that name was less popular. 
        This is a measure that makes sure that we plot the names that were popular both as male name and as 
        female name and that the ensuing plot of the delta goes well above zero and well below zero for each name."""
        
        dfy = self.data.groupby(['name', 'year', 'sex'], 
                            as_index = False).sum().pivot_table(values = "births", 
                                                                columns = "sex", 
                                                                index = ["name", "year"]).fillna(0).reset_index()
        dfy["delta"] = dfy.F - dfy.M


        df = dfy[['name', 'delta']].groupby("name").agg(["max", "min"]).reset_index()
        df["check"] = 0
        df.loc[(df["delta", "max"] > 0) & (df["delta", "min"] < 0), "check"] = 1
        df = df[df.check==1]

        df = pd.DataFrame({"name" : df.name, "max" : df[("delta", "max")], "min" : df[("delta", "min")].apply(abs)})
        df["sorter"] = df[["max", "min"]].min(axis = 1)

        result = pd.merge(df.sort_values(by="sorter", ascending = False).head(n),
                  dfy[["name", "year", "delta"]], on='name')

        t = "Top {} names that flipped over the years \nPositive values indicate female prevalence".format(n)
        result.pivot(index = "year", columns = "name", values = "delta").fillna(0).plot(title=t,figsize=(12,6))


    def NameJump(self, name="Bill", year=2000, state='all',sex='both') :
        """This function is needed for the second part, when we develop a story based on movies and their effect
        on the popularity of names.
        For a given name and year, returns the change in popularity in the subsequent year"""
        def add_prop(group):
            group['prop'] = round(100.0*group.births / group.births.sum(), 4)
            return group
        if state != "all" or sex != "both" :
            df =self.data
            if state != "all" :
                df = df[df.state==state]
            if sex != "both" :
                df = df[df.sex==sex]
            df = df.groupby(["name","year"],as_index=False).sum()
            df = df.groupby('year').apply(add_prop)
        else :
            df = self.grouped_data
        df = df[((df["year"]==year) | (df["year"]==year+1)) & (df["name"] == name)]
        return df.iloc[1].prop - df.iloc[0].prop

def mainchar1(castlist) :
    """Return first name of character with top billing in movie credits"""
    try : 
        return json.loads(castlist)[0]['character'].split()[0]
    except :
        return np.nan
    
def mainchar2(castlist) :
    """Return first name of character with secondary billing in movie credits"""
    try : 
        return json.loads(castlist)[1]['character'].split()[0]
    except :
        return np.nan
    
def MovieJump(name, year, babynames) :
    """Apply babynames method NameJump to given name and year to measure increase in popularity 
    for given name in year immediately following the film's release"""
    try :
        return babynames.NameJump(name=name, year=year)
    except :
        return np.nan

m['char1'] = m.cast.apply(mainchar1)

The first 5 Harry Potter movies were released in 2001, 2002, 2004, 2005, 2007. We see that in the year after the release of each movie there was an increase in the popularity of the name Hermione. The falls happen exactly one year after the years in which a movie was not released (2003 and 2006). Again, this could be totally spurious, but the data tracks the release dates very well.
An interesting fact is that the release of the first book (1997) did not have an effect on the popularity of the name Hermione. It's not impossible to rationalize this: movies reach a broader audience than books. Furthermore, there might be young kids who are too young to read the book but old enough to watch the movie and who could "put pressure" on their parents to name their baby sister "Hermione" after seeing the movie.
